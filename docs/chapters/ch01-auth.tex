\chapter{Authentication \& Token Management}

\section{Feature Overview}
Authentication service with JWT token management (access tokens for stateless verification, refresh tokens for persistent storage and logout).

\section{Function: \texttt{register(data: RegisterInput)}}

\subsection{Entry Point}
\begin{verbatim}
register(data: RegisterInput) → { user, accessToken, refreshToken }
Parameters: firstName, lastName, email, phone, password, role?
\end{verbatim}

\subsection{Decision Points \& Branches}

\subsubsection{Decision 1: Email Uniqueness Check}
\begin{itemize}
    \item \textbf{Condition}: \texttt{existingEmail = prisma.user.findUnique(\{where: \{email\}\})}
    \item \textbf{Branch A (True)}: Email exists $\rightarrow$ throw Error(\textit{``Email already exists''})
    \item \textbf{Branch B (False)}: Email unique $\rightarrow$ continue
\end{itemize}

\subsubsection{Decision 2: Phone Uniqueness Check}
\begin{itemize}
    \item \textbf{Condition}: \texttt{existingPhone = prisma.user.findUnique(\{where: \{phone\}\})}
    \item \textbf{Branch A (True)}: Phone exists $\rightarrow$ throw Error(\textit{``Phone number already exists''})
    \item \textbf{Branch B (False)}: Phone unique $\rightarrow$ continue
\end{itemize}

\subsubsection{Decision 3-7: Processing Steps}
\begin{enumerate}
    \item Password Hashing (deterministic)
    \item User Creation (should succeed with valid data)
    \item Token Generation (deterministic)
    \item Delete Old Tokens (idempotent)
    \item Store Refresh Token (deterministic)
\end{enumerate}

1 & Email unique $\rightarrow$ Phone unique $\rightarrow$ User created $\rightarrow$ Success & Register $\rightarrow$ Hash password $\rightarrow$ Create user $\rightarrow$ Generate tokens $\rightarrow$ Delete old $\rightarrow$ Store refresh token $\rightarrow$ Return user+tokens & User created with tokens & 201 \\
\hline

\section{Function: \texttt{login(data: LoginInput)}}

\subsection{Entry Point}
\begin{verbatim}
login(data: LoginInput) → { user, accessToken, refreshToken }
Parameters: email, password
\end{verbatim}

\subsection{Decision Points \& Branches}

\subsubsection{Decision 1: User Lookup}
\begin{itemize}
    \item \textbf{Condition}: \texttt{user = prisma.user.findUnique(\{where: \{email\}\})}
    \item \textbf{Branch A (True)}: User found $\rightarrow$ continue
    \item \textbf{Branch B (False)}: User not found $\rightarrow$ throw Error(\textit{``Invalid credentials''})
\end{itemize}

\subsubsection{Decision 2: Password Verification}
\begin{itemize}
    \item \textbf{Condition}: \texttt{isPasswordValid = verifyPassword(password, user.password)}
    \item \textbf{Branch A (True)}: Password valid $\rightarrow$ continue
    \item \textbf{Branch B (False)}: Password invalid $\rightarrow$ throw Error(\textit{``Invalid credentials''})
\end{itemize}

\subsubsection{Decision 3: Account Active Check}
\begin{itemize}
    \item \textbf{Condition}: \texttt{user.isActive}
    \item \textbf{Branch A (True)}: User active $\rightarrow$ continue
    \item \textbf{Branch B (False)}: User deactivated $\rightarrow$ throw Error(\textit{``User account is deactivated''})
\end{itemize}
